    def uses_switch(choice, *args)
      external_name = extract_switch_raw_name(args)
      external_names[choice] = '--' + external_name
      args = change_name_to_switch(external_name, args)
      @parser.on(*args) do | value |
        self[choice] = value.to_s
      end
    end
