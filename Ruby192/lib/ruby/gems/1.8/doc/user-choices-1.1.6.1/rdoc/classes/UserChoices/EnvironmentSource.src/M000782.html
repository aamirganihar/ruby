    def with_prefix(prefix)
      matches = ENV.collect do | env_var, ignored_value |
        if /^#{prefix}(.+)/ =~ env_var
          [$1.to_inputable_sym, env_var]
        end
      end
      @external_names.merge!(Hash[*matches.compact.flatten])
      self
    end
