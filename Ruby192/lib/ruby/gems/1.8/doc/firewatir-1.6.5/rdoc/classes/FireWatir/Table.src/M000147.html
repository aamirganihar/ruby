    def row_values(row)
      assert_exists
      arr_rows = rows
      cells = arr_rows[row - 1].cells
      values = Array.new(cells.length)
      for i in 0..cells.length - 1 do
        values[i] = cells[i].to_s
      end
      return values
    end
