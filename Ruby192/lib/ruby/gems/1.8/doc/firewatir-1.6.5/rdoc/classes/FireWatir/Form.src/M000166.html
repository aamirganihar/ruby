    def locate
      # Get form using xpath.
      case @how
      when :jssh_name
        @element_name = @what
      when :xpath
        @element_name = element_by_xpath(@container, @what)
      else
        @element_name = locate_tagged_element("form", @how, @what)
      end
      @o = self
    end
