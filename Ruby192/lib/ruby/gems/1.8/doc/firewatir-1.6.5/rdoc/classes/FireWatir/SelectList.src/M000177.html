    def selected_options
      assert_exists
      returnArray = []
      #element.log "There are #{@o.length} items"
      each do |thisItem|
        #puts "#{thisItem.selected}"
        if thisItem.selected
          #element.log "Item ( #{thisItem.text} ) is selected"
          returnArray << thisItem.text
        end
      end
      return returnArray
    end
