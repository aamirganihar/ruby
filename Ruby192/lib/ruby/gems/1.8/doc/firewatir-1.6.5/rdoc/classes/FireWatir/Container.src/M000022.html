    def show_all_objects
      puts "-----------Objects in the current context-------------" 
      locate if respond_to?(:locate)
      elements = Document.new(self).all
      puts elements.length
      elements.each  do |n|
        puts n.tagName
        puts n.to_s
        puts "------------------------------------------" 
      end
      puts "Total number of objects in the current context :    #{elements.length}"
      return elements

    end
