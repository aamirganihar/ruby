    def text()
      assert_exists
      element = (element_type == "HTMLFrameElement") ? "body" : element_object
      return_value = js_eval("#{element}.textContent.replace(/\\xA0/g, ' ').replace(/\\s+/g, ' ')").strip
      @@current_level = 0
      return return_value
    end
