  def with_command_args(string)
    begin
      old_argv = ARGV.dup
      ARGV.replace(string.split)
      yield
    rescue SystemExit => ex
      replacement = StandardError.new(ex.message)
      replacement.set_backtrace(ex.backtrace)
      raise replacement
    ensure
      ARGV.replace(old_argv)
    end
  end
