  def set_test_paths(given)
    return if S4tUtils.const_defined?("PACKAGE_ROOT")

    path = File.expand_path(given)
    i = path.index("/test/")
    S4tUtils.const_set("PACKAGE_ROOT", path[0...i])
    $:.unshift("#{PACKAGE_ROOT}/lib")
    $:.unshift("#{PACKAGE_ROOT}")
  end
