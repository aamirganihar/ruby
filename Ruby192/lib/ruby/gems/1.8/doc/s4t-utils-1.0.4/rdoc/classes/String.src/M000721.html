  def without_pretty_indentation(margin=nil)
    s = self.dup
    # Remove initial blank line.
    s.sub!(/\A[ \t]*\n/, "")
    # Get rid of the margin, if it's specified.
    unless margin.nil?
      margin_re = Regexp.escape(margin || "")
      margin_re = /^[ \t]*#{margin_re} ?/
      s.gsub!(margin_re, "")
    end
    # Remove trailing whitespace on each line
    s.gsub!(/[ \t]+$/, "")
    s
  end
