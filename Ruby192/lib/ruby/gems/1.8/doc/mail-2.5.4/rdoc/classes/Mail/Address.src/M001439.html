    def format
      parse unless @parsed
      case
      when tree.nil?
        ''
      when display_name
        [quote_phrase(display_name), "<#{address}>", format_comments].compact.join(" ")
      when address
        [address, format_comments].compact.join(" ")
      else
        tree.text_value
      end
    end
