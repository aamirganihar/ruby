    def ready_to_send!
      identify_and_set_transfer_encoding
      parts.sort!([ "text/plain", "text/enriched", "text/html", "multipart/alternative" ])
      parts.each do |part|
        part.transport_encoding = transport_encoding
        part.ready_to_send!
      end
      add_required_fields
    end
