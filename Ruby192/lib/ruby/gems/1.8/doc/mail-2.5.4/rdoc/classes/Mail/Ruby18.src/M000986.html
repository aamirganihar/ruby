    def Ruby18.b_value_decode(str)
      match = str.match(/\=\?(.+)?\?[Bb]\?(.+)?\?\=/m)
      if match
        encoding = match[1]
        str = Ruby18.decode_base64(match[2])
        str = Iconv.conv('UTF-8//IGNORE', fix_encoding(encoding), str)
      end
      str
    end
