        def xls_data
                line = "1"
                data = []
                        while @worksheet.Range("a#{line}")['Value']
                           data << @worksheet.Range("a#{line}:#{number_columns}#{line}")['Value']
                           line.succ!
                        end
                
                header = data[0].flatten
                dataObject = Struct.new("DataObject", *header)
                data.shift
                data_array = []
                data.each do |line|
                        data_array << dataObject.new(*line.flatten)
                end
        @excel.Quit
        @excel = nil
        GC.start
        data_array
        end
