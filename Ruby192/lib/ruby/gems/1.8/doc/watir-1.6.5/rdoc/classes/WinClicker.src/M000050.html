  def getTextValueForFileNameField(parenthWnd) 
    f = getHandleOfControl(parenthWnd, "ComboBox", 1)
    if f == -1 then
      # unable to find the first combobox
      return nil
    else
      # we have the control and now
      # can send it some messages
      return getWinText(f )
    end
  end
