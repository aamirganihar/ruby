    def speed= how_fast
      case how_fast
      when :zippy then
              @typingspeed = 0
              @pause_after_wait = 0.01
              @type_keys = false
              @speed = :fast
      when :fast then
              @typingspeed = 0
              @pause_after_wait = 0.01
              @type_keys = true
              @speed = :fast
      when :slow then
              @typingspeed = 0.08
              @pause_after_wait = 0.1
              @type_keys = true
              @speed = :slow
      else
        raise ArgumentError, "Invalid speed: #{how_fast}"
      end
    end
