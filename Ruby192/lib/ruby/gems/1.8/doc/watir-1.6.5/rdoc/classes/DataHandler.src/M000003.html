        def csv_data
                csv_data = File.readlines(@datafile)
                header = csv_data[0].split(",")
                header.each{|z| z.chomp!}
                data = Struct.new("Data", *header)
                csv_data.shift
                data_array = []
                csv_data.each do |line|
                        line.each{|z| line.chomp!}
                        data_array << data.new(*line.split(","))
                end
        return data_array
        end
