        def number_columns
                column = last_column = "a"
                while @worksheet.Range("a1:#{column}1")['Value']
                        if !@worksheet.Range("#{column}1")['Value']
                                last_column = column.pred
                                break
                        end
                        column.succ!
                end
        last_column
        end
