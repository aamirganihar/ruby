        def get_line
                @anchor = getLength
                if @inputAdded == 0
                        Thread.stop
                end
                @inputAdded -= 1
                retval = @input[0].gets
                # don't print every prompt for multiline input
                @im.print_prompt = (@inputAdded == 0) 
                return retval
        end
