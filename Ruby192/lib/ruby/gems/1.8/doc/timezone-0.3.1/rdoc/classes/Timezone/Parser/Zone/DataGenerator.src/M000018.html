      def generate(zone)
        zones = Timezone::Parser.zones[zone].to_a

        return if zones.empty?

        @@data[zone] = zones
          .each_cons(2)
          .inject(update(zones.first)) do |set, (previous, zone)|
            update(zone, set, previous && previous.end_date)
        end
      end
