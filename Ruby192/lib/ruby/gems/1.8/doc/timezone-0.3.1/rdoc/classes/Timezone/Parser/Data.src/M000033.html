      def initialize(start_date, end_date, zone, rule = NilRule.new)
        @dst = rule.dst?
        @offset = parse_offset(zone, rule)
        @name = parse_name(zone.format, rule.letter)
        @utime = rule.utime?
        @start_date = parse_start_date(start_date)
        self.end_date = end_date
      end
